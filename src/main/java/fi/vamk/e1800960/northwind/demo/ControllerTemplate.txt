
package fi.vamk.e1800960.northwind.demo.Controller;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;

import org.springframework.web.bind.annotation.PathVariable;

import org.springframework.web.bind.annotation.RequestBody;

import org.springframework.web.bind.annotation.RequestMapping;

import org.springframework.web.bind.annotation.RequestMethod;

import org.springframework.web.bind.annotation.RequestParam;

import org.springframework.web.bind.annotation.ResponseBody;

import org.springframework.web.bind.annotation.RestController;

import fi.vamk.e1800960.northwind.demo.Repo.*;



@RestController

public class EntityController {

	@Autowired

	private EntityRepository repository;
	
    @RequestMapping(value = "/Entity" , method = RequestMethod.POST)

    public @ResponseBody Entity create(@RequestBody Entity item) {

    	return repository.save(item);

    }  

	

    @RequestMapping("/Entity")

    public Iterable<Entity> municipalities(@RequestParam(value="name", defaultValue="World") String name) {

        return repository.findAll();

    }    

    

    @RequestMapping("/Entity/{id}")

    public Optional<Entity> get(@PathVariable("id") int id) {
        return repository.findById(id);
    }    



    @RequestMapping(value = "/Entity" , method = RequestMethod.PUT)

    public @ResponseBody Entity update(@RequestBody Entity item) {

    	return repository.save(item);

    } 

    

    @RequestMapping(value = "/Entity" , method = RequestMethod.DELETE)

    public void delete(@RequestBody Entity item) {

    	repository.delete(item);

        //do business logic

    } 

}
